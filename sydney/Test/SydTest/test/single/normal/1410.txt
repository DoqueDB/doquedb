# 表を作成
Command "create table T (N int, C ntext, L language)";

#insert
CreatePreparedCommand "ins" "insert into T (N, C, L) values (?, ?, ?)";
PreparedCommand "ins" [0, textsjisfile "..\\..\\doc\\kenpou.txt", "ja"];

["%の%", "ja", "%the%", "en"];

["%の%", "ja", "%the%", "en"];

	hint 'delayed, inverted=(normalized=true, indexing=dual,
	tokenizer=DUAL:JAP:ALL:1 @NORMRSCID:1 @UNARSCID:1)'";

Command "select N from T where C like ? language ? and C like ? language ?"
["%の%", "ja", "%the%", "en"];

["%の%", "ja", "%the%", "en"];

PreparedCommand "sel" ["%EU%", "en"];

# verify
